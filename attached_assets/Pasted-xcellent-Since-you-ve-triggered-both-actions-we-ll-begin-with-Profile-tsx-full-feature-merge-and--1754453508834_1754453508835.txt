"use client";

import { useSession } from "next-auth/react";
import { useEffect, useState } from "react";
import EmojiReactions from "./EmojiReactions";
import PlaylistCreator from "./PlaylistCreator";
import CapsuleGrid from "./CapsuleGrid";
import SMRIBadge from "./SMRIBadge";

export default function Profile() {
  const { data: session } = useSession();
  const [stats, setStats] = useState(null);

  useEffect(() => {
    if (!session?.user?.id) return;
    fetch(`/api/profile/stats/${session.user.id}`)
      .then((res) => res.json())
      .then(setStats);
  }, [session?.user?.id]);

  if (!stats) {
    return (
      <div className="p-6 bg-[#0d0d0d] text-white rounded-xl shadow-inner animate-pulse space-y-4">
        <div className="h-5 w-1/2 bg-[#222] rounded"></div>
        <div className="h-5 w-1/3 bg-[#222] rounded"></div>
        <div className="h-5 w-2/3 bg-[#222] rounded"></div>
      </div>
    );
  }

  return (
    <div className="p-6 text-white space-y-6">
      <div className="flex items-center justify-between">
        <h1 className="text-3xl font-bold">👤 {session.user.name || "Your Profile"}</h1>
        <SMRIBadge userId={session.user.id} />
      </div>

      <div className="grid grid-cols-3 gap-4 bg-black/60 rounded-xl p-4 shadow">
        <p>👁 Total Views: {stats.totalViews}</p>
        <p>❤️ Likes: {stats.totalLikes}</p>
        <p>👥 Followers: {stats.followers}</p>
      </div>

      <EmojiReactions capsuleId={"profile-global"} userId={session.user.id} />

      <PlaylistCreator userId={session.user.id} />

      <CapsuleGrid userId={session.user.id} />
    </div>
  );
}
