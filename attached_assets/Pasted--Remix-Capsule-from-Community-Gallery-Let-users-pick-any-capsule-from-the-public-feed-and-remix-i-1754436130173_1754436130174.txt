🏆 Capsule Remix Contest Mode
Enable a “Contest Week” feature where all remixes are entered into a leaderboard.

1. Supabase Table
sql
Copy
Edit
create table if not exists remix_contests (
  id uuid primary key default gen_random_uuid(),
  name text,
  description text,
  start_time timestamptz,
  end_time timestamptz,
  is_active boolean default true
);

create table if not exists remix_contest_entries (
  id uuid primary key default gen_random_uuid(),
  remix_id uuid references capsule_remixes(id),
  contest_id uuid references remix_contests(id),
  votes int default 0
);
2. API: Submit Entry

function RemixLeaderboard({ contestId }) {
  const { data } = useSWR(`/api/remix/contest/leaderboard?contestId=${contestId}`, fetcher);

  return (
    <div className="p-4 bg-black rounded-lg text-white">
      <h3 className="text-xl mb-2">🏆 Remix Leaderboard</h3>
      {data?.entries.map((entry, i) => (
        <div key={entry.id} className="flex justify-between border-b border-gray-700 py-2">
          <span>#{i + 1} Remix by {entry.user_name}</span>
          <span>{entry.votes} 🔥 votes</span>
        </div>
      ))}
    </div>
  );
}
