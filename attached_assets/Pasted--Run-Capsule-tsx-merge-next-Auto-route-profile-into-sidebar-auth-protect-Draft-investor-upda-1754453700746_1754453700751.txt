"use client";

import { useEffect, useState } from "react";
import CapsuleTrendGraph from "./CapsuleTrendGraph";
import CapsuleValue from "./CapsuleValue";
import ReactEmojiBar from "./ReactEmojiBar";

export default function Capsule({ capsuleId }) {
  const [capsuleStats, setCapsuleStats] = useState(null);
  const [trendData, setTrendData] = useState([]);

  useEffect(() => {
    if (!capsuleId) return;

    fetch(`/api/capsule/stats/${capsuleId}`).then(res => res.json()).then(setCapsuleStats);
    fetch(`/api/capsule/trend/${capsuleId}`).then(res => res.json()).then(setTrendData);
  }, [capsuleId]);

  if (!capsuleStats) return <div>Loading capsule details...</div>;

  return (
    <div className="space-y-6 p-6 text-white">
      <h2 className="text-3xl font-bold">🧬 Capsule Details</h2>

      <CapsuleValue capsuleStats={capsuleStats} />

      <ReactEmojiBar capsuleId={capsuleId} />

      <CapsuleTrendGraph trendData={trendData} />

      <a
        href={`/api/capsule/audit/${capsuleId}?format=csv`}
        className="bg-blue-600 text-white px-4 py-2 rounded shadow hover:bg-blue-700"
      >
        ⬇️ Export Audit CSV
      </a>
    </div>
  );
}
