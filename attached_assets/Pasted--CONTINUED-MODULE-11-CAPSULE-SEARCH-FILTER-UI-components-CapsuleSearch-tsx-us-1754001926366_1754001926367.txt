// components/CapsuleSearch.tsx
"use client";
import { useState } from 'react';

export default function CapsuleSearch({ onSearch }) {
  const [query, setQuery] = useState('');
  return (
    <div className="flex gap-2 mb-4">
      <input
        type="text"
        className="w-full border px-3 py-2 rounded shadow"
        placeholder="Search title, grief score, or author..."
        value={query}
        onChange={(e) => setQuery(e.target.value)}
      />
      <button
        onClick={() => onSearch(query)}
        className="bg-indigo-600 text-white px-4 py-2 rounded shadow"
      >
        Search
      </button>
    </div>
  );
}

// components/FilterPanel.tsx
export default function FilterPanel({ filters, setFilters }) {
  return (
    <div className="flex gap-4 flex-wrap text-sm mb-4">
      <select
        onChange={(e) => setFilters({ ...filters, sortBy: e.target.value })}
        className="border px-2 py-1 rounded"
      >
        <option value="recent">Most Recent</option>
        <option value="yield">Highest Yield</option>
        <option value="score">Grief Score</option>
      </select>
      <select
        onChange={(e) => setFilters({ ...filters, tier: e.target.value })}
        className="border px-2 py-1 rounded"
      >
        <option value="all">All Tiers</option>
        <option value="sealed">Veritas Sealed</option>
        <option value="open">Unverified</option>
      </select>
    </div>
  );
}
